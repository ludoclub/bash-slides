Les boucles: until


#!/bin/bash
i=1
until [ $i –gt 100 ]; do
 echo $i*7=$(($i*7))
 i=$(($i+1))
done

# !/bin/bash
select choix in "entree 1" "entree 2" "q"; do
 case $choix in
    "entree 1" | "entree 2") echo "Vous avez choisi" $choix ;;
    "q") echo "Au revoir…"
 break ;;
 *) echo "Saisie non valide !"
 esac
done
